# Namespace Search-and-Replace Map
# ORDER MATTERS — apply longest match first to avoid partial replacements
#
# Format: OLD_NAMESPACE → NEW_NAMESPACE
#
# Phase 1: Namespace migration
# ============================

# Memory navigators (must come before Memory)
PLang.Runtime2.Memory.Navigators → PLang.Runtime2.Engine.Memory.Navigators

# Core → Engine root
PLang.Runtime2.Core → PLang.Runtime2.Engine

# Context
PLang.Runtime2.Context → PLang.Runtime2.Engine.Context

# Memory (after Navigators already handled)
PLang.Runtime2.Memory → PLang.Runtime2.Engine.Memory

# IO → Channels
PLang.Runtime2.IO → PLang.Runtime2.Engine.Channels

# Errors
PLang.Runtime2.Errors → PLang.Runtime2.Engine.Errors

# Serialization → Serializers
PLang.Runtime2.Serialization → PLang.Runtime2.Engine.Serializers

# Utility
PLang.Runtime2.Utility → PLang.Runtime2.Engine.Utility

# Parsing
PLang.Runtime2.Parsing → PLang.Runtime2.Engine.Parsing

# Mapping
PLang.Runtime2.Mapping → PLang.Runtime2.Engine.Mapping

# modules/ infrastructure files only → Libraries
# These 8 files move from modules/ root to Libraries/:
#   IClass.cs, ICodeGenerated.cs, IContext.cs,
#   ActionAttribute.cs, DefaultAttribute.cs, VariableNameAttribute.cs,
#   Libraries.cs, Library.cs
# Their namespace changes:
PLang.Runtime2.modules (in infrastructure files only) → PLang.Runtime2.Engine.Libraries

# modules/ handler subfolders stay as modules
# No namespace change needed IF they already use PLang.Runtime2.modules
# Just need folder move: modules/{handler}/ → Engine/modules/{handler}/
# The Discover() call updates:
#   "PLang.Runtime2.modules" → "PLang.Runtime2.Engine.modules"
